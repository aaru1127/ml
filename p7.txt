import numpy as np, pandas as pd
import matplotlib.pyplot as plt
from sklearn.datasets import load_diabetes
from sklearn.linear_model import LinearRegression as LR
from sklearn.preprocessing import PolynomialFeatures as PF
from sklearn.model_selection import train_test_split as split
from sklearn.metrics import mean_squared_error as mse, r2_score as r2

def lin():
    x, y = load_diabetes(return_X_y=True)
    x_train, x_test, y_train, y_test = split(x, y, test_size=0.2, random_state=42)

    m = LR().fit(x_train, y_train)
    y_pred = m.predict(x_test)

    plt.figure(figsize=(10,6))
    plt.scatter(y_test, y_pred, alpha=0.6)
    plt.plot([y_test.min(), y_test.max()], [y_test.min(), y_test.max()], 'r', lw=2)
    plt.xlabel("True Values (Boston Housing)")
    plt.ylabel("Predicted Values")
    plt.title("Linear Regression on Boston Housing Dataset")
    plt.grid() 
    plt.show()

    print("Linear Regression Results:")
    print(f"Mean Squared Error: {mse(y_test, y_pred):.2f}")
    print(f"R^2 Score: {r2(y_test, y_pred):.2f}")

def poly():
    df = pd.read_csv("https://raw.githubusercontent.com/mwaskom/seaborn-data/master/mpg.csv").dropna()
    x, y = df[["horsepower"]].values, df["mpg"].values
    x_train, x_test, y_train, y_test = split(x, y, test_size=0.2, random_state=42)

    pf = PF(3)
    m = LR().fit(pf.fit_transform(x_train), y_train)
    y_pred = m.predict(pf.transform(x_test))

    xs = np.sort(x_test.flatten())
    ys_fit = m.predict(pf.transform(xs.reshape(-1,1)))

    plt.figure(figsize=(10,6))
    plt.scatter(x_test, y_test, alpha=0.6, label="True values")
    plt.plot(xs, ys_fit, 'r', lw=2, label="Polynomial fit (degree=3)")
    plt.xlabel("Horsepower"); plt.ylabel("Miles Per Gallon (MPG)")
    plt.title("Polynomial Regression on Auto MPG Dataset")
    plt.legend(); plt.grid(); plt.show()

    print("Polynomial Regression Results (Degree=3):")

    print(f"R^2 Score: {r2(y_test, y_pred):.2f}")

lin()
poly()
